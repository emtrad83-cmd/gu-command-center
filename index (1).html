🗄️ Supabase — prospects table

Run this SQL once in your Supabase dashboard to enable Pipeline sync:

create table prospects (id bigint generated always as identity primary key, name text, contact text, mentor text, stage text, warmth text, tension_point text, sample_status text, six_w_status text, last_conversation text, follow_up_date text, notes text, reach_out_count int default 0, is_core boolean default false, created_at timestamptz default now());

create table prospect_coaching (id bigint generated always as identity primary key, prospect_id bigint, archetype text, stage text, stage_number int, emotional_temp text, tension_detected text[], power_frame text, coaching text, next_move text, generated_message text, created_at timestamptz default now());
